document       := meta? assets? components? pages+
pages          := "page" ident block+
block          := NEWLINE INDENT (kvpair | section | shape | group | frame)+ DEDENT
section        := ("assets" | "components" | "meta") block
frame          := "frame" ident block
group          := "group" ident block
shape          := (rect|ellipse|diamond|text|icon|image|arrow|line|polyline|sticky|freehand) ident block
kvpair         := key ":" value
key            := ident ( "." ident )*
value          := string | number | bool | enum | list | ref
ref            := "#" ident
